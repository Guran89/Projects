--Räkna Cf (utan självciteringar) för publikationer, ta med publiceringsår och titel

SELECT c.pubyear, c.title,
	CASE
		WHEN py = 2010 THEN cwts.cx5/cwts.fcsx5 --Vilka publiceringsår som används. I det här fallet 2010-2013
		WHEN py = 2011 THEN cwts.cx4/cwts.fcsx4
		WHEN py = 2012 THEN cwts.cx3/cwts.fcsx3
		WHEN py = 2013 THEN cwts.cx2/cwts.fcsx2
		ELSE cwts.cx5/cwts.fcsx5
	END AS cxf
FROM cwts
INNER JOIN cross_isi_cpl AS cic ON cwts.isi_id = cic.isi_id
INNER JOIN uplmain AS c ON cic.pubid = c.pubid
WHERE c.pubid IN () --PubID mellan parenteser
AND c.pubyear BETWEEN 2010 AND 2013 --Ändra här till de år som är relevanta
