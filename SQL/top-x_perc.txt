--Kod för att ta fram Top-X% för en viss författare.
--Första koden tar fram ANTAL artiklar.

SELECT COUNT(cwts.percx5)
FROM cwts
INNER JOIN cross_isi_cpl AS cic ON cwts.isi_id = cic.isi_id
INNER JOIN uplmain AS c ON cic.cpl_pubid = c.pubid
INNER JOIN person_publ AS pp ON c.dbid = pp.dbid
WHERE pp.personid = 269 --Ändra till aktuellt personID här
AND cwts.percx5 > 80 --70 = Top-20%, 80 = Top-10%, 90 = Top-5%


--För att få fram ANDEL artiklar, använd följande kod.
--Dela sedan resultatet från första koden med resultatet från nedanstående för att få fram andel.

SELECT COUNT(cwts.percx5)
FROM cwts
INNER JOIN cross_isi_cpl AS cic ON cwts.isi_id = cic.isi_id
