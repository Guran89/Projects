--Tar fram poster där ett visst orcid är knutet till 2 eller fler personID
SELECT DISTINCT personid, p.firstname, p.lastname, p.external_user_id, p.orcid
FROM persons AS p
INNER JOIN 
	(SELECT p.orcid, COUNT(*), p.external_user_id
	FROM persons AS p
	GROUP BY p.orcid, p.external_user_id
	HAVING COUNT(p.orcid) > 1) AS temp ON p.external_user_id = temp.external_user_id
INNER JOIN person_publ AS pp ON p.personid = pp.personid
INNER JOIN uplmain AS c ON pp.dbid = c.dbid
INNER JOIN dept_pers_publ AS dpp ON pp.pd_id = dpp.pd_id
INNER JOIN dept_instance AS di ON dpp.deptid = di.deptid
WHERE di.instanceid = 1
AND c.pubyear BETWEEN 2008 AND 2016
ORDER BY p.orcid
